177bool ArchiveWriter::addFile(const std::filesystem::path &base, const std::filesystem::path &file) {
178 auto &&pathname = (base / file.filename()).string();
179
180 struct archive_entry *entry = archive_entry_new();
181 archive_entry_set_pathname(entry, pathname.c_str());
182 archive_entry_set_size(entry, static_cast<int64_t>(std::filesystem::file_size(file)));
183 archive_entry_set_filetype(entry, AE_IFREG);
184 archive_write_header(XX, entry);
185
186 int64_t len;
187 auto stream = io::File::create(file.string(), BINARY_READ_EXISTED);
188 char buffer[4096];
189 while ((len = stream->read(buffer, 4096)) > 0) {
190 if (archive_write_data(XX, buffer, static_cast<size_t>(len)) != len) {
191 stream->close();
192 stream.reset();
193 archive_entry_free(entry);
194 return false;
195 }
196 }
197 stream->close();
198 stream.reset();
199 archive_entry_free(entry);
200 return true;
201}

207bool ArchiveWriter::addDirectory(const std::filesystem::path &base, const std::filesystem::path &dir) {
208 for (const auto &item: std::filesystem::directory_iterator(dir)) {
209 if (item.is_directory()) {
210 if (ArchiveWriter::addDirectory(base / item.path().filename(), item)) {
211 continue;
212 } else {
213 return false;
214 }
215 }
216
217 // 压缩包中的相对路径
218 auto &&pathname = (base / item.path().filename()).string();
219 // 原始文件系统中的绝对路径
220 auto &&file_path = item.path().string();
221
222 struct archive_entry *entry = archive_entry_new();
223 archive_entry_set_pathname(entry, pathname.c_str());
224 archive_entry_set_size(entry, static_cast<int64_t>(std::filesystem::file_size(file_path)));
225 archive_entry_set_filetype(entry, AE_IFREG);
226 archive_write_header(XX, entry);
227
228 int64_t len;
229 auto file = io::File::create(file_path, BINARY_READ_EXISTED);
230 char buffer[4096];
231 while ((len = file->read(buffer, 4096)) > 0) {
232 if (archive_write_data(XX, buffer, static_cast<size_t>(len)) != len) {
233 file->close();
234 file.reset();
235 archive_entry_free(entry);
236 return false;
237 }
238 }
239 file->close();
240 file.reset();
241 archive_entry_free(entry);
242 }
243 return true;
244}

246bool ArchiveWriter::addStream(const std::filesystem::path &path, io::InputStream *input, size_t exp) {
247 if (nullptr == input) {
248 return false;
249 }
250
251 auto &&pathname = path.string();
252
253 struct archive_entry *entry = archive_entry_new();
254 archive_entry_set_pathname(entry, pathname.c_str());
255 archive_entry_set_size(entry, static_cast<int64_t>(exp));
256 archive_entry_set_filetype(entry, AE_IFREG);
257 archive_entry_set_is_data_encrypted(entry, 1);
258 archive_write_header(XX, entry);
259
260 while (exp) {
261 char buffer[4096];
262 const auto LEN = input->read(buffer, std::min<size_t>(exp, 4096));
263 if (LEN <= 0) {
264 break;
265 }
266 if (const auto WRITE = archive_write_data(XX, buffer, static_cast<size_t>(LEN));WRITE != LEN) {
267 break;
268 } else {
269 exp -= static_cast<size_t>(LEN);
270 }
271 }
272 archive_entry_free(entry);
273 if (exp) {
274 return false;
275 } else {
276 return true;
277 }
278}
