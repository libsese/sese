25IPCChannel::Ptr IPCChannel::create(const std::string &name, size_t buffer_size) {
26 auto sem_name = name + "00Semaphore";
27 auto mem_name = name + "00Memory";
28
29 auto sem = Semaphore::create(sem_name);
30 if (!sem) {
31 return nullptr;
32 }
33
34 sem->lock();
35 auto mem = SharedMemory::create(mem_name.c_str(), buffer_size + sizeof(MemInfo));
36 if (!mem) {
37 return nullptr;
38 }
39
40 auto buffer = static_cast<char *>(mem->getBuffer());
41 auto mem_info = reinterpret_cast<MemInfo *>(buffer + 0);
42 memset(mem_info, 0, sizeof(MemInfo));
43 mem_info->total_size = buffer_size;
44 mem_info->tail = 0;
45
46 auto first_message = reinterpret_cast<MemInfo *>(buffer + sizeof(uint32_t));
47 memset(first_message, 0, 4);
48
49 sem->unlock();
50
51 auto ipc = new IPCChannel();
52 ipc->mem_info = mem_info;
53 ipc->buffer = buffer + sizeof(MemInfo);
54 ipc->semaphore = std::move(sem);
55 ipc->shared_memory = std::move(mem);
56 return std::unique_ptr<IPCChannel>(ipc);
57}

86bool IPCChannel::write(uint32_t id, const void *data, uint32_t length) {
87 Locker locker(*semaphore);
88 // 确认共享内存空间是否足够入队一条信息
89 if (mem_info->tail + HEADER_SIZE + length > mem_info->total_size + 4) {
90 return false;
91 }
92
93 // 拷贝信息至共享内存
94 auto buf = static_cast<char *>(buffer);
95 memcpy(buf + mem_info->tail + 0, &id, sizeof(id));
96 memcpy(buf + mem_info->tail + 4, &length, sizeof(length));
97 memcpy(buf + mem_info->tail + 8, data, length);
98
99 // 更新头部内存信息
100 mem_info->tail += 8 + length;
101 return true;
102}

110std::list<Message> IPCChannel::read(uint32_t id) {
111 std::list<Message> list;
112 std::list<MessageIterator> iterators;
113 Locker locker(*semaphore);
114
115 auto buf = static_cast<char *>(this->buffer);
116 MessageIterator message{};
117 while (true) {
118 memcpy(&message.id, buf + 0, sizeof(uint32_t));
119 if (message.id == 0) {
120 break;
121 } else {
122 memcpy(&message.length, buf + 4, sizeof(uint32_t));
123 if (message.length) {
124 message.buffer = buf + 8;
125 iterators.push_back(message);
126 buf += 8 + message.length;
127 } else {
128 message.buffer = nullptr;
129 iterators.push_back(message);
130 buf += 8;
131 }
132 }
133 }
134
135 size_t size = 0;
136 buf = static_cast<char *>(this->buffer);
137 for (auto && iterator = iterators.begin(); iterator != iterators.end(); ) {
138 if (iterator->id == id) {
139 list.emplace_back(std::string(static_cast<char *>(iterator->buffer), iterator->length));
140 iterator = iterators.erase(iterator);
141 } else {
142 memcpy(buf + 0, &iterator->id, sizeof(uint32_t));
143 memcpy(buf + 4, &iterator->length, sizeof(uint32_t));
144 memmove(buf + 8, iterator->buffer, iterator->length);
145 buf += 8 + iterator->length;
146 size += 8 + iterator->length;
147 iterator++;
148 }
149 }
150 memset(buf, 0, sizeof(uint32_t));
151 this->mem_info->tail = size;
152
153 return list;
