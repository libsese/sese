123void IOCPService::onAccept(int fd) {
124 if (sese::net::Socket::setNonblocking(fd)) {
125 sese::net::Socket::close(fd);
126 }
127
128 const auto CTX = new Context;
129 CTX->self = master;
130 CTX->client = this;
131 CTX->fd = fd;
132
133 if (master->getServCtx()) {
134 const auto SERVER_SSL = static_cast<SSL_CTX *>(master->getServCtx()->getContext());
135 SSL *client_ssl = SSL_new(SERVER_SSL);
136 SSL_set_fd(client_ssl, fd);
137 SSL_set_accept_state(client_ssl);
138
139 while (true) {
140 if (const auto RT = SSL_do_handshake(client_ssl); RT <= 0) {
141 if (const auto ERR = SSL_get_error(client_ssl, RT); ERR != SSL_ERROR_WANT_READ && ERR != SSL_ERROR_WANT_WRITE) {
142 SSL_free(client_ssl);
143 sese::net::Socket::close(fd);
144 delete CTX;
145 return;
146 }
147 } else {
148 CTX->ssl = client_ssl;
149 SSL_set_mode(client_ssl, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
150 const uint8_t *data = nullptr;
151 uint32_t data_length;
152 SSL_get0_alpn_selected(client_ssl, &data, &data_length);
153 onAlpnGet(CTX, data, data_length);
154 break;
155 }
156 }
157 }
158
159 onAcceptCompleted(CTX);
160}

162void IOCPService::onRead(sese::event::BaseEvent *event) {
163 const auto CTX = static_cast<Context *>(event->data);
164
165 onPreRead(CTX);
166
167 size_t len = 0;
168 while (true) {
169 char buf[MTU_VALUE];
170 if (const auto L = read(static_cast<int>(CTX->fd), buf, MTU_VALUE, CTX->ssl); L <= 0) {
171 if (L == 0 && len == 0) {
172 releaseContext(CTX);
173 break;
174 } else {
175 onReadCompleted(CTX);
176 break;
177 }
178 } else {
179 CTX->recv.write(buf, L);
180 len += static_cast<size_t>(L);
181 }
182 }
183}

185void IOCPService::onWrite(sese::event::BaseEvent *event) {
186 if (const auto CTX = static_cast<Context *>(event->data); CTX->isConn) {
187 CTX->isConn = false;
188 auto ssl = static_cast<SSL *>(CTX->ssl);
189 if (CTX->ssl) {
190 SSL_set_connect_state(ssl);
191 // GCOVR_EXCL_START
192 while (true) {
193 if (const auto RT = SSL_do_handshake(ssl); RT <= 0) {
194 // err is SSL_ERROR_WANT_READ or SSL_ERROR_WANT_WRITE
195 if (const auto ERR = SSL_get_error(ssl, RT); ERR != SSL_ERROR_WANT_READ && ERR != SSL_ERROR_WANT_WRITE) {
196 SSL_free(ssl);
197 CTX->ssl = nullptr;
198 ssl = nullptr;
199 break;
200 }
201 } else {
202 break;
203 }
204 }
205 // GCOVR_EXCL_STOP
206 if (ssl == nullptr) {
207 releaseContext(CTX);
208 return;
209 } else {
210 SSL_set_mode(ssl, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
211 const uint8_t *data = nullptr;
212 uint32_t data_length;
213 SSL_get0_alpn_selected(ssl, &data, &data_length);
214 onAlpnGet(CTX, data, data_length);
215 }
216 }
217 onConnected(CTX);
218 } else {
219 while (true) {
220 char buf[MTU_VALUE];
221 const auto LEN = CTX->send.peek(buf, MTU_VALUE);
222 if (LEN == 0) {
223 CTX->send.freeCapacity();
224 onWriteCompleted(CTX);
225 break;
226 }
227 if (const auto L = write(static_cast<int>(CTX->fd), buf, LEN, CTX->ssl); L <= 0) {
228 if (const auto ERR = sese::net::getNetworkError(); ERR == EWOULDBLOCK || ERR == EINTR) {
229 postWrite(CTX);
230 break;
231 } else {
232 releaseContext(CTX);
233 break;
234 }
235 } else {
236 CTX->send.trunc(L);
237 }
238 }
239 }
240}

336void IOCPServer::postConnect(const net::IPAddress::Ptr &to, const security::SSLContext::Ptr &cli_ctx, void *data) {
337 const auto SOCK = sese::net::Socket::socket(to->getFamily(), SOCK_STREAM, IPPROTO_IP);
338 sese::net::Socket::setNonblocking(SOCK);
339
340 if (connect(SOCK, to->getRawAddress(), to->getRawAddressLength())) {
341 if (const auto ERR = sese::net::getNetworkError(); ERR != EINPROGRESS) {
342 return;
343 }
344 }
345
346 const auto CTX = new Context;
347 CTX->fd = SOCK;
348 CTX->self = this;
349 CTX->isConn = true;
350 CTX->data = data;
351 if (cli_ctx) {
352 CTX->ssl = SSL_new(static_cast<SSL_CTX *>(cli_ctx->getContext()));
353 // printf("ssl new %p\n", ctx->ssl);
354 SSL_set_fd(static_cast<SSL *>(CTX->ssl), static_cast<int>(SOCK));
355 SSL_set_alpn_protos(static_cast<SSL *>(CTX->ssl), reinterpret_cast<const unsigned char *>(clientProtos.c_str()), static_cast<unsigned>(clientProtos.length()));
356 }
357
358 balanceLoader.dispatchSocket(SOCK, CTX);
359}
