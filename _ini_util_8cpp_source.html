6IniUtil::IniConfigObject IniUtil::parse(IniUtil::InputStream *input) noexcept {
7 auto reader = io::InputStreamReader(input);

9 auto config = std::make_unique<IniConfig>();
10 auto current = &config->defSection;
11
12 while (true) {
13 std::string line = reader.readLine();
14 if (line.empty()) {
15 break;
16 }
17
18 if (line[0] == '[') {
19 // new section
20 auto pos = line.find_last_of(']');
21 if (pos == std::string::npos) {
22 continue;
23 }
24 auto iterator = config->sectionMap.emplace(line.substr(1, pos - 1), std::map<std::string, std::string>{});
25 current = &iterator.first->second;
26 } else if (line[0] == '#') {
27 // comment
28 continue;
29 } else {
30 // current section
31 auto pos = line.find_first_of('=');
32 if (pos == std::string::npos) {
33 continue;
34 }
35 current->emplace(line.substr(0, pos), line.substr(pos + 1));
36 }
37 }
38
39 return config;

48bool IniUtil::unparse(const IniUtil::IniConfigObject &target, IniUtil::OutputStream *output) noexcept {
49 for (auto &&item: target->defSection) {
50 if (!unparsePair(item, output)) {
