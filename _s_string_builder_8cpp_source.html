73 size_t count = sstr::getStringLengthFromUTF8String(str);
74 size_t new_size = count + _size;

124void SStringBuilder::trim() {
125 size_t i = 0; // 头部空格数
126 size_t j = 0; // 尾部空格数
127 bool flag = true;
128 for (auto n = 0; n < _size; n++) {
129 if (flag && ' ' == _data[n]) {
130 i++;
131 } else {
132 flag = false;
133 }
134
135 if (' ' == _data[_size - 1 - n]) {
136 j++;
137 } else {
138 break;
139 }
140 }
141
142 // 头尾空白相接
143 if (i + j == _size) {
144 _data[0] = 0;
145 return;
146 }
147 // 正常处理
148 else {
149 size_t len = _size - j - i;
150 for (size_t n = 0; n < len; n++) {
151 _data[n] = _data[i + n];
152 }
153 _data[len] = 0;
154 }
155}

286void SStringBuilder::replace(size_t begin, size_t len, const char *u8str) {
287 if (begin + 1 > _size) return;
288
289 auto str = SStringView(u8str);
290 auto chars = str.toChars();
291 auto char_size = chars.size();
292 auto new_size = _size - len + char_size;
293 // 需要扩容
294 if (new_size > _cap) {
295 reserve((new_size / BLOCK_SIZE + 1) * BLOCK_SIZE);
296 }
297
298 // 为插入内容提供空间
299 if (char_size > len) {
300 RightShiftElement(_data, _size, begin + len, char_size - len);
301 } else if (char_size < len) {
302 LeftShiftElement(_data, _size, begin + char_size, len - char_size);
303 }
304
305 // 直接替换
306 for (size_t i = 0; i < char_size; i++) {
307 _data[begin + i] = (uint32_t) chars[i];
308 }
309 _size = new_size;
310}
