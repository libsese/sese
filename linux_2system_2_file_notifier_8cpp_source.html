45void FileNotifier::loopNonblocking() noexcept {
46 auto proc = [this]() {
47 fd_set set;
48 struct timeval timeout {
49 1, 0
50 };
51 char buffer[1024];
52 while (!is_shutdown) {
53 FD_ZERO(&set);
54 FD_SET(inotify_fd, &set);
55 select(FD_SETSIZE, &set, nullptr, nullptr, &timeout);
56 if (FD_ISSET(inotify_fd, &set)) {
57 auto p_event = (inotify_event *) &buffer;
58 auto len = read(inotify_fd, buffer, sizeof(buffer));
59 auto times = (len / sizeof(inotify_event)) - 1;
60 char *from_string = nullptr;
61 while (times) {
62 if (p_event->wd != watch_fd) {
63 p_event++;
64 times--;
65 continue;
66 } else {
67 if (p_event->mask & IN_CREATE) {
68 option->onCreate({p_event->name});
69 } else if (p_event->mask & IN_MODIFY) {
70 option->onModify({p_event->name});
71 } else if (p_event->mask & IN_DELETE) {
72 option->onDelete({p_event->name});
73 } else if (p_event->mask & IN_MOVED_FROM) {
74 from_string = p_event->name;
75 } else if (p_event->mask & IN_MOVED_TO) {
76 if (from_string) {
77 option->onMove({from_string}, {p_event->name});
78 from_string = nullptr;
79 }
80 }
81 times--;

83 }
84 }
85 }
