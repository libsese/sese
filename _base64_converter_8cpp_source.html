21void Base64Converter::encode(InputStream *src, OutputStream *dest) {
22 unsigned char buffer[3]{0};
23 unsigned char result[4]{0};
24
25 int64_t len;
26 while ((len = src->read(buffer, 3)) != 0) {
27 if (len == 2) {
28 buffer[2] = 0b00000000;
29 } else if (len == 1) {
30 buffer[1] = 0b00000000;
31 buffer[2] = 0b00000000;
32 }
33
34 unsigned char bits = 0b00000000;
35 // 1
36 bits = (buffer[0] & 0b11111100) >> 2;
37 ::encode(bits, result[0]);
38 // 2
39 bits = (buffer[0] & 0b00000011) << 4 | (buffer[1] & 0b11110000) >> 4;
40 ::encode(bits, result[1]);
41 // 3
42 if (len >= 2) {
43 bits = (buffer[1] & 0b00001111) << 2 | (buffer[2] & 0b11000000) >> 6;
44 ::encode(bits, result[2]);
45
46 // 4
47 if (len == 3) {
48 bits = (buffer[2] & 0b00111111);
49 ::encode(bits, result[3]);
50 }
51 }
52 dest->write(result, len + 1);
53 if (len != 3) dest->write("==", len == 1 ? 2 : 1);
54 }
55}

63void Base64Converter::decode(InputStream *src, OutputStream *dest) {
64 unsigned char buffer[4]{0};
65
66 int64_t len;
67 while ((len = src->read(buffer, 4)) != 0) {
68 int offset = 0;
69 for (auto i = 0; i < len; i++) {
70 // 此处已是逻辑完备的
71 // GCOVR_EXCL_START
72 if (buffer[i] >= 'A' && buffer[i] <= 'Z') {
73 buffer[i] -= 65;
74 } else if (buffer[i] >= 'a' && buffer[i] <= 'z') {
75 buffer[i] -= 71;
76 } else if (buffer[i] >= '0' && buffer[i] <= '9') {
77 buffer[i] += 4;
78 } else if (buffer[i] == '+' || buffer[i] == '-') {
79 buffer[i] = 62;
80 } else if (buffer[i] == '/' || buffer[i] == '_') {
81 buffer[i] = 63;
82 } else {
83 buffer[i] = 0;
84 offset -= 1;
85 }
86 // GCOVR_EXCL_STOP
87 }
88
89 len += offset;
90 // 1
91 if (len >= 2) {
92 buffer[0] = (buffer[0] << 2) | (buffer[1] & 0b00110000) >> 4;
93
94 // 2
95 if (len >= 3) {
96 buffer[1] = (buffer[1] & 0b00001111) << 4 | (buffer[2] & 0b00111100) >> 2;
97
98 // 3
99 if (len == 4) {
100 buffer[2] = (buffer[2] & 0b00000011) << 6 | buffer[3];
101 }
102 }
103 }
104
105 dest->write(buffer, len - 1);
106 }

115bool Base64Converter::encodeInteger(size_t num, OutputStream *output) noexcept {
116 if (num == 0) {
117 output->write(&base62_code_page[0], 1);
118 return true;
119 }
120
121 std::vector<unsigned char> vector;
122 auto base = strlen((const char *) base62_code_page);
123 while (num > 0) {
124 std::div_t res = div((int) num, (int) base);
125 num = res.quot;
126 vector.push_back(base62_code_page[res.rem]);
127 }
128
129 std::reverse(vector.begin(), vector.end());
130 return output->write(vector.data(), vector.size()) == vector.size();
131}

133int64_t Base64Converter::decodeBuffer(const unsigned char *buffer, size_t size) noexcept {
134 auto base = strlen((const char *) base62_code_page);
135 auto page = std::string_view((const char *) base62_code_page, base);
136 int64_t num = 0;
137 for (auto idx = 0; idx < size; ++idx) {
138 auto power = (size - (idx + 1));
139 auto i = page.find((char) (buffer[idx]));
140 if (i == std::string::npos) return -1;
141 num += i * static_cast<int64_t>(std::pow(base, power)); // NOLINT
142 }
143 return num;
144}

146bool Base64Converter::encodeBase62(InputStream *input, OutputStream *output) noexcept {
147 unsigned char buffer[4]{};
148
149 int64_t len;
150 while ((len = input->read(buffer + 1, 3)) != 0) {
151 if (len == 2) {
152 buffer[3] = buffer[2];
153 buffer[2] = buffer[1];
154 buffer[1] = 0;
155 } else if (len == 1) {
156 buffer[3] = buffer[1];
157 buffer[2] = 0;
158 buffer[1] = 0;
159 }
160 // buffer[0] always be zero
161 auto num = *(uint32_t *) &buffer;
162 num = FromBigEndian32(num);
163 if (!encodeInteger(num, output)) {
164 return false;
165 }
166 }

172 unsigned char buffer[4]{};
