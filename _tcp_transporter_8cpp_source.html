41void sese::service::TcpTransporter::onAccept(int fd) {
42 SSL *client_ssl;
43 // GCOVR_EXCL_START
44 if (sese::net::Socket::setNonblocking(fd)) {
45 sese::net::Socket::close(fd);
46 }
47
48 // auto conn = new TcpConnection;
49 auto conn = config->createConnection();
50
51 // GCOVR_EXCL_STOP
52 if (config->servCtx) {
53 client_ssl = SSL_new((SSL_CTX *) config->servCtx->getContext());
54 SSL_set_fd(client_ssl, (int) fd);
55 SSL_set_accept_state(client_ssl);
56
57 // GCOVR_EXCL_START
58 while (true) {
59 auto rt = SSL_do_handshake(client_ssl);
60 if (rt <= 0) {
61 auto err = SSL_get_error(client_ssl, rt);
62 if (err != SSL_ERROR_WANT_READ && err != SSL_ERROR_WANT_WRITE) {
63 SSL_free(client_ssl);
64 sese::net::Socket::close(fd);
65 // delete conn;
66 config->freeConnection(conn);
67 return;
68 }
69 } else {
70 conn->ssl = client_ssl;
71 // 此选项允许 OpenSSL 在尝试重试 SSL_write 时使用不完全相同的 buffer 参数
72 // https://stackoverflow.com/questions/2997218/why-am-i-getting-error1409f07fssl-routinesssl3-write-pending-bad-write-retr
73 SSL_set_mode(client_ssl, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
74 // ALPN Callback
75 const uint8_t *data = nullptr;
76 uint32_t data_length;
77 SSL_get0_alpn_selected(client_ssl, &data, &data_length);
78 onProcAlpnGet(conn, data, data_length);
79 break;
80 }
81 }
82 // GCOVR_EXCL_STOP
83 }
84
85 if (config->keepalive) {
86 conn->timeoutEvent = createTimeoutEvent(fd, conn, config->keepalive);
87 }
88 conn->event = createEventEx(fd, EVENT_READ, conn);
89}

126void sese::service::TcpTransporter::onWrite(sese::event::BaseEvent *event) {
127 auto conn = (TcpConnection *) event->data;
128 char buf[MTU_VALUE];
129 while (true) {
130 auto len = conn->buffer2write.peek(buf, MTU_VALUE);
131 if (len == 0) {
132 conn->buffer2write.freeCapacity();
133 break;
134 }
135 auto l = write(event->fd, buf, len, conn->ssl);
136 if (l <= 0) {
137 auto err = sese::net::getNetworkError();
138 if (err == EWOULDBLOCK || err == EINTR) {
139 conn->event->events &= ~EVENT_READ;
140 conn->event->events |= EVENT_WRITE;
141 this->setEvent(event);
142 break;
143 } else {
144 if (conn->timeoutEvent) {
145 TimerableService::freeTimeoutEvent(conn->timeoutEvent);
146 }
147 onProcClose(conn);
148 if (config->servCtx) {
149 SSL_free((SSL *) conn->ssl);
150 }
151 sese::net::Socket::close(event->fd);
152 // delete conn;
153 config->freeConnection(conn);
154 this->freeEventEx(event);
155 break;
156 }
157 } else {
158 conn->buffer2write.trunc(l);
159 }
160 }
161
162 conn->event->events |= EVENT_READ;
163 conn->event->events &= ~EVENT_WRITE;
164 this->setEvent(event);
165}
