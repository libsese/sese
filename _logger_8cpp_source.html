39void Logger::log(const Event::Ptr &event) noexcept {
40 std::string content = formatter->dump(event);
41 if (builtInAppender->getLevel() <= event->getLevel()) {
42 setbuf(stdout, nullptr);
43 // 此处逻辑本身就是完备的
44 switch (event->getLevel()) {
45 case Level::DEBUG:
46 ConsoleAppender::setDebugColor();
47 break;
48 case Level::INFO:
49 ConsoleAppender::setInfoColor();
50 break;
51 case Level::WARN:
52 ConsoleAppender::setWarnColor();
53 break;
54 case Level::ERR:
55 ConsoleAppender::setErrorColor();
56 break;
57 }
58 builtInAppender->dump(content.c_str(), content.length());
59 ConsoleAppender::setCleanColor();
60 fflush(stdout);
61 }
62
63 for (auto &appender: appenderVector) {
64 if (appender->getLevel() <= event->getLevel()) {
65 appender->dump(content.c_str(), content.length());
66 }
67 }
68}
