26bool ArchiveReader::extract(const std::filesystem::path &src_path, const std::filesystem::path &dest_path,
27 const std::string &pwd) {
28 const auto SRC = src_path.string();
29 auto dest = dest_path.string();
30
31 struct archive *a = archive_read_new();
32 // archive_read_support_compression_all(XX);
33 archive_read_support_filter_all(a);
34 archive_read_support_format_all(a);
35
36 if (!pwd.empty()) {
37 if (ARCHIVE_OK != archive_read_add_passphrase(a, pwd.c_str())) {
38 archive_read_free(a);
39 return false;
40 }
41 }
42
43 if (ARCHIVE_OK != archive_read_open_filename(a, SRC.c_str(), 4096)) {
44 archive_read_free(a);
45 return false;
46 }
47
48 struct archive_entry *entry{};
49 while (archive_read_next_header(a, &entry) == ARCHIVE_OK) {
50 std::string filename = archive_entry_pathname(entry);
51
52 auto exp = archive_entry_size(entry);
53 if (const auto TYPE = archive_entry_filetype(entry); AE_IFDIR == TYPE) {
54 if ('/' == filename.at(filename.length() - 1)) {
55 filename = filename.substr(0, filename.length() - 1);
56 }
57 if (auto path = dest_path / filename; std::filesystem::exists(path)) {
58 if (!std::filesystem::is_directory(path)) {
59 archive_read_free(a);
60 return false;
61 }
62 } else {
63 if (!std::filesystem::create_directories(path)) {
64 archive_read_free(a);
65 return false;
66 }
67 }
68 continue;
69 }
70
71 if (auto parent = (dest_path / filename).parent_path(); !std::filesystem::exists(parent)) {
72 if (!std::filesystem::create_directories(parent)) {
73 archive_read_free(a);
74 return false;
75 }
76 }
77
78 auto filepath = (dest_path / filename).string();
79 const auto FILE = File::create(filepath, BINARY_WRITE_CREATE_TRUNC);
80 while (exp) {
81 char buffer[4096];
82 const auto LEN = archive_read_data(a, buffer, std::min<size_t>(exp, 4096));
83 if (LEN <= 0) {
84 break;
85 }
86 auto write = FILE->write(buffer, static_cast<size_t>(LEN));
87 if (write != LEN) {
88 break;
89 } else {
90 exp -= LEN;
91 }
92 }
93
94 FILE->flush();
95 FILE->close();
96 if (exp) {
97 archive_read_free(a);
98 return false;
99 }
100 }
101
102 archive_read_free(a);
103 return true;
104}

140bool ArchiveReader::extract(const ArchiveReader::ExtractCallback &callback) {
141 // archive_read_support_compression_all(XX);
142 archive_read_support_filter_all(XX);
143 archive_read_support_format_all(XX);
144 if (ARCHIVE_OK != archive_read_open(
145 XX,
146 this,
147 open,
148 read,
149 close
150 )) {
151 return false;
152 }
153
154 struct archive_entry *entry{};
155 auto archive_input_stream = ArchiveInputStream(archive);
156 while (archive_read_next_header(XX, &entry) == ARCHIVE_OK) {
157 std::string filename = archive_entry_pathname(entry);
158
159 const auto SIZE = archive_entry_size(entry);
160 const auto TYPE = archive_entry_filetype(entry);
161
162 callback(filename, Config::fromValue(TYPE), &archive_input_stream, SIZE);
163 }
164 return true;
165}
