86template<class SERVICE>
87bool sese::service::SystemBalanceLoader::init(std::function<SERVICE *()> creator) noexcept {
88 if (address == nullptr) return false;
89
90 sese::net::ReusableSocket reusable_socket(address);
91 for (size_t i = 0; i < threads; ++i) {
92 auto sub_socket = reusable_socket.makeRawSocket();
93 if (sub_socket == -1) {
94 goto freeSocket;
95 }
96 if (0 != sese::net::Socket::setNonblocking(sub_socket)) {
97 goto freeSocket;
98 }
99 if (0 != sese::net::Socket::listen(sub_socket, 32)) {
100 goto freeSocket;
101 }
102 socketVector.emplace_back(sub_socket);
103 }
104
105 for (size_t i = 0; i < threads; ++i) {
106 auto event = creator();
107 if (event == nullptr) {
108 goto freeEvent;
109 }
110 event->setListenFd(static_cast<int>(socketVector[i]));
111 if (!event->init()) {
112 delete event;
113 goto freeEvent;
114 } else {
115 eventLoopVector.emplace_back(event);
116 }
117 }
118
119 return true;
120
121freeEvent:
122 for (decltype(auto) event_loop: eventLoopVector) {
123 delete event_loop;
124 }
125 eventLoopVector.clear();
126
127freeSocket:
128 for (decltype(auto) sub_socket: socketVector) {
129 sese::net::Socket::close(sub_socket);
130 }
131 socketVector.clear();
132 return false;
133}
