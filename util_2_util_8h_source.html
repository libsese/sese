129template<class T>
130size_t number2StringLength(T num, size_t radix = 10) {
131 static_assert(!std::is_same_v<T, double>, "Must be an integer");
132 static_assert(!std::is_same_v<T, float>, "Must be an integer");
133 size_t length = 0;
134 if (num == 0) return 1;
135 if (num < 0) {
136 length += 1;
137 num *= -1;
138 }
139 do {
140 num /= static_cast<T>(radix);
141 length += 1;
142 } while (num > 0);
143 return length;
144}

154 static_assert(std::is_same_v<T, double> || std::is_same_v<T, float>, "Must be a floating number");

157 length += 1 + placeholder;

177template <typename T>
178bool isAdditionOverflow(T a, T b) {
179 if constexpr (std::is_unsigned_v<T>) {
180 return a > std::numeric_limits<T>::max() - b;
181 } else {
182 if ((b > 0 && a > std::numeric_limits<T>::max() - b) ||
183 (b < 0 && a < std::numeric_limits<T>::min() - b)) {
184 return true;
185 }
186 }
187 return false;
188}
