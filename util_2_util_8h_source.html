129template<class T>
130size_t number2StringLength(T num, size_t radix = 10) {
131 static_assert(!std::is_same_v<T, double>, "Must be an integer");
132 static_assert(!std::is_same_v<T, float>, "Must be an integer");
133 size_t length = 0;
134 if (num == 0) return 1;
135 if (num < 0) {
136 length += 1;
137 num *= -1;
138 }
139 do {
140 num /= static_cast<T>(radix);
141 length += 1;
142 } while (num > 0);
143 return length;
144}

153size_t floating2StringLength(T num, uint16_t placeholder) {
154 static_assert(std::is_same_v<T, double> || std::is_same_v<T, float>, "Must be a floating number");
155 size_t length = number2StringLength(static_cast<int64_t>(num));
156 if (placeholder > 0) {
157 length += 1 + placeholder;
158 }
159 return length;
160}
