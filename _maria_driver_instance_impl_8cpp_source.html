29PreparedStatement::Ptr impl::MariaDriverInstanceImpl::createStatement(const char *sql) noexcept {
30 MYSQL_STMT *stmt = mysql_stmt_init(conn);
31 if (stmt == nullptr) return nullptr;
32
33 size_t len = strlen(sql);
34 if (mysql_stmt_prepare(stmt, sql, (unsigned long) len)) return nullptr;
35
36 // 此处需要手动计算 '?' 个数
37 size_t count = 0;
38 for (size_t i = 0; i < len; ++i) {
39 if (sql[i] == '?') count++;
40 }
41
42 // 获取结果集元数据，对于非 DQL 而言这可能是 nullptr，不应作为错误处理
43 auto meta = mysql_stmt_result_metadata(stmt);
44
45 return std::make_unique<impl::MariaPreparedStatementImpl>(stmt, meta, count);
46}
