48void sese::event::KqueueEventLoop::dispatch(uint32_t time) {
49 struct kevent events[MAX_EVENT_SIZE]{};
50 auto div = ldiv(time, 1000);
51 struct timespec timeout {
52 div.quot, div.rem * 1000000
53 };
54
55 auto number_of_fds = ::kevent(kqueue, nullptr, 0, events, MAX_EVENT_SIZE, &timeout);
56 if (-1 == number_of_fds) return;
57 for (int i = 0; i < number_of_fds; ++i) {
58 auto event = reinterpret_cast<BaseEvent *>(events[i].udata);
59 if (events[i].ident == listenFd) {
60 auto client = accept(listenFd, nullptr, nullptr);
61 if (client == -1) continue;
62 onAccept(client);
63 continue;
64 } else {
65 if (events[i].flags & EV_ERROR && event->events & EVENT_ERROR) {
66 onError(event);
67 continue;
68 }
69 if (events[i].filter == EVFILT_READ) {
70 if (event->events & EVENT_READ) {
71 onRead(event);
72 }
73
74 if ((events[i].flags & EV_EOF) && handleClose) {
75 onClose(event);
76 continue;
77 }
78 } else if (events[i].filter == EVFILT_WRITE) {
79 if ((events[i].flags & EV_EOF) && handleClose) {
80 onClose(event);
81 continue;
82 }
83
84 if (event->events & EVENT_WRITE) {
85 onWrite(reinterpret_cast<BaseEvent *>(events[i].udata));
86 }
87 }
88 }
89 }
