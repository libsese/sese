19#define ByteSwap16(i16) ((i16 & 0x00FF) << 8 | (i16 & 0xFF00) >> 8)
20#define ByteSwap32(i32) ((i32 & 0x000000FF) << 24 | (i32 & 0x0000FF00) << 8 | (i32 & 0x00FF0000) >> 8 | (i32 & 0xFF000000) >> 24)
21#define ByteSwap64(i64) \
22 ((i64 & 0x00000000000000FF) << 56 | \
23 (i64 & 0x000000000000FF00) << 40 | \
24 (i64 & 0x0000000000FF0000) << 24 | \
25 (i64 & 0x00000000FF000000) << 8 | \
26 (i64 & 0x000000FF00000000) >> 8 | \
27 (i64 & 0x0000FF0000000000) >> 24 | \
28 (i64 & 0x00FF000000000000) >> 40 | \
29 (i64 & 0xFF00000000000000) >> 56)

266inline float ToBigEndian32(float t) {
267 ASSERT_FLOAT_SIZE;
268#ifdef SESE_LITTLE_ENDIAN
269 int32_t tmp;
270 memcpy(&tmp, &t, sizeof(float));
271 ToBigEndian32<int32_t>(tmp);
272 memcpy(&t, &tmp, sizeof(float));
273#endif
274 return t;
