44inline std::string requestTypeToString(RequestType request_type) {
45 switch (request_type) {
46 case RequestType::OPTIONS:
47 return "OPTIONS";
48 case RequestType::GET:
49 return "GET";
50 case RequestType::POST:
51 return "POST";
52 case RequestType::HEAD:
53 return "HEAD";
54 case RequestType::PUT:
55 return "PUT";
56 case RequestType::DELETE:
57 return "DELETE";
58 case RequestType::TRACE:
59 return "TRACE";
60 case RequestType::CONNECT:
61 return "CONNECT";
62 case RequestType::ANOTHER:
63 default:
64 return "ANOTHER";
65 }
66}
