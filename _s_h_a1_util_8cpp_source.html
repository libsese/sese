15inline static uint32_t rol(const uint32_t VALUE, const size_t BITS) {
16 return (VALUE << BITS) | (VALUE >> (32 - BITS));
17}

19inline static uint32_t blk(const uint32_t block[BLOCK_INTS], const size_t I) {
20 return rol(block[(I + 13) & 15] ^ block[(I + 8) & 15] ^ block[(I + 2) & 15] ^ block[I], 1);
21}

23/*
24 * (R0+R1), R2, R3, R4 are the different operations used in SHA1

26inline static void R0(const uint32_t block[BLOCK_INTS], const uint32_t V, uint32_t &w, const uint32_t X, const uint32_t Y, uint32_t &z, const size_t I) {
27 z += ((w & (X ^ Y)) ^ Y) + block[I] + 0x5a827999 + rol(V, 5);
28 w = rol(w, 30);
29}

31inline static void R1(uint32_t block[BLOCK_INTS], const uint32_t V, uint32_t &w, const uint32_t X, const uint32_t Y, uint32_t &z, const size_t I) {
32 block[I] = blk(block, I);
33 z += ((w & (X ^ Y)) ^ Y) + block[I] + 0x5a827999 + rol(V, 5);
34 w = rol(w, 30);
35}

37inline static void R2(uint32_t block[BLOCK_INTS], const uint32_t V, uint32_t &w, const uint32_t X, const uint32_t Y, uint32_t &z, const size_t I) {
38 block[I] = blk(block, I);
39 z += (w ^ X ^ Y) + block[I] + 0x6ed9eba1 + rol(V, 5);
40 w = rol(w, 30);
41}

43inline static void R3(uint32_t block[BLOCK_INTS], const uint32_t V, uint32_t &w, const uint32_t X, const uint32_t Y, uint32_t &z, const size_t I) {
44 block[I] = blk(block, I);
45 z += (((w | X) & Y) | (w & X)) + block[I] + 0x8f1bbcdc + rol(V, 5);
46 w = rol(w, 30);
47}

49inline static void R4(uint32_t block[BLOCK_INTS], const uint32_t V, uint32_t &w, const uint32_t X, const uint32_t Y, uint32_t &z, const size_t I) {
50 block[I] = blk(block, I);
51 z += (w ^ X ^ Y) + block[I] + 0xca62c1d6 + rol(V, 5);
52 w = rol(w, 30);

64 R0(block, a, b, c, d, e, 0);
65 R0(block, e, a, b, c, d, 1);
66 R0(block, d, e, a, b, c, 2);
67 R0(block, c, d, e, a, b, 3);

77 R0(block, c, d, e, a, b, 13);

153inline static void buffer_to_block(const uint8_t *buffer, uint32_t block[BLOCK_INTS]) {
154 /* Convert the std::string (byte buffer) to an uint32_t array (MSB) */
155 for (size_t i = 0; i < BLOCK_INTS; i++) {
156 block[i] = (buffer[4 * i + 3] & 0xff) | (buffer[4 * i + 2] & 0xff) << 8 | (buffer[4 * i + 1] & 0xff) << 16 | (buffer[4 * i + 0] & 0xff) << 24;
157 }
158}

166void SHA1Util::encode(InputStream *input, OutputStream *output) noexcept {
167 SHA1Context ctx;
168 uint8_t buffer[64];
169 uint32_t block[64];
170
171 int64_t size;
172 while ((size = input->read(buffer, 64)) == 64) {
173 ctx.total += size;
174 buffer_to_block(buffer, block);
175 transform(&ctx, block);
176 }
177
178 // 处理尾部
179 if (size > 55) {
180 ctx.total += size;
181 buffer[size] = 0x80;
182 memset(buffer + size + 1, 0, 63 - size);
183
184 buffer_to_block(buffer, block);
185 transform(&ctx, block);
186
187 // 最后填充一次
188 uint64_t total = ToBigEndian64(ctx.total * 8);
189 memset(buffer, 0, 56);
190 memcpy(buffer + 56, &total, 8);
191
192 buffer_to_block(buffer, block);
193 transform(&ctx, block);
194 } else {
195 ctx.total += size;
196 uint64_t total = ToBigEndian64(ctx.total * 8);
197 buffer[size] = 0x80;
198 memset(buffer + size + 1, 0, 55 - size);
199 memcpy(buffer + 56, &total, 8);
200
201 buffer_to_block(buffer, block);
202 transform(&ctx, block);
203 }
204
205 for (unsigned int &i: ctx.h) {
206 i = FromBigEndian32(i);
207 }
208 output->write(ctx.h, sizeof(ctx.h));
209}

229std::unique_ptr<char[]> SHA1Util::encode(InputStream *input, bool is_cap) noexcept {
230 ByteBuilder dest(64);
231 encode(input, &dest);
232 unsigned char buffer[40];
233 auto rt = std::unique_ptr<char[]>(new char[41]);
234 dest.read(buffer, 20);
235 for (auto i = 0; i < 20; ++i) {
236 rt[i * 2 + 1] = toChar(buffer[i] % 0x10, is_cap);
237 rt[i * 2 + 0] = toChar(buffer[i] / 0x10, is_cap);
238 }
239 rt[40] = 0;
240 return rt;
241}
