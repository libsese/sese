13bool HttpUtil::getLine(InputStream *source, StringBuilder &builder) noexcept {
14 char ch;
15 for (size_t i = 0; i < HTTP_MAX_SINGLE_LINE + 1; i++) {
16 size_t size = source->read(&ch, 1);
17 if (size != 1) {
18 return false;
19 }
20
21 if (ch != '\r') {
22 builder.append(ch);
23 } else {
24 // 剩下的 '\n'，读取的结果已经不重要

173bool HttpUtil::recvHeader(InputStream *source, StringBuilder &builder, Header *header, bool is_resp) noexcept {
174 CookieMap::Ptr cookies;
175 if (is_resp) {
176 cookies = std::make_shared<CookieMap>();
177 }
178
179 while (true) {
180 if (!getLine(source, builder)) {
181 return false;
182 }
183
184 // \r\n${无内容}\r\n
185 if (builder.empty()) {
186 break;
187 } else {
188 // auto pair = builder.split(": ");
189 // if (pair.size() != 2) return false;
190 auto line = builder.toString();
191 builder.clear();
192 auto pos = line.find_first_of(": ");
193 if (pos == std::string::npos) {
194 return false;
195 }
196 auto key = line.substr(0, pos);
197 auto value = line.substr(pos + 2);
198 if (is_resp) {
199 if (strcasecmp(key.c_str(), "Set-Cookie") == 0) {
200 auto cookie = parseFromSetCookie(value);
201 if (cookie != nullptr) {
202 cookies->add(cookie);
203 }
204 } else {
205 header->set(key, value);
206 }
207 } else {
208 if (strcasecmp(key.c_str(), "Cookie") == 0) {
209 cookies = parseFromCookie(value);
210 } else {
211 header->set(key, value);
212 }
213 }
214 }
215 }
216 header->setCookies(cookies);
217 return true;
218}

223bool HttpUtil::sendHeader(OutputStream *dest, Header *header, bool is_resp) noexcept {
224 size_t len;
225 for (const auto &pair: *header) {
226 len = pair.first.length();
227 WRITE(pair.first.c_str(), len); // GCOVR_EXCL_LINE
228 WRITE(": ", 2); // GCOVR_EXCL_LINE
229 len = pair.second.length();
230 WRITE(pair.second.c_str(), len); // GCOVR_EXCL_LINE
231 WRITE("\r\n", 2); // GCOVR_EXCL_LINE
232 }
233
234 auto cookies = header->getCookies();
235 if (cookies != nullptr) {
236 if (is_resp) {
237 if (!sendSetCookie(dest, cookies)) return false;
238 } else {
239 if (!sendCookie(dest, cookies)) return false;
240 }
241 }
242
243
244 if (-1 == dest->write("\r\n", 2)) return false; // GCOVR_EXCL_LINE
245 return true;

248bool HttpUtil::sendSetCookie(OutputStream *dest, const CookieMap::Ptr &cookies) noexcept {
249 for (decltype(auto) cookie: *cookies) {
250 WRITE("Set-Cookie: ", 12); // GCOVR_EXCL_LINE
251 const std::string &name = cookie.first;
252 const std::string &value = cookie.second->getValue();
253 WRITE(name.c_str(), name.size()); // GCOVR_EXCL_LINE
254 WRITE("=", 1); // GCOVR_EXCL_LINE
255 WRITE(value.c_str(), value.size()); // GCOVR_EXCL_LINE
256
257 const std::string &path = cookie.second->getPath();
258 if (!path.empty()) {
259 WRITE("; Path=", 7); // GCOVR_EXCL_LINE
260 WRITE(path.c_str(), path.size()); // GCOVR_EXCL_LINE
261 }
262
263 const std::string &domain = cookie.second->getDomain();
264 if (!domain.empty()) {
265 WRITE("; Domain=", 9); // GCOVR_EXCL_LINE
266 WRITE(domain.c_str(), domain.size()); // GCOVR_EXCL_LINE
267 }
268
269 uint64_t max_age = cookie.second->getMaxAge();
270 if (max_age > 0) {
271 WRITE("; Max-Age=", 10); // GCOVR_EXCL_LINE
272 auto age = std::to_string(max_age);
273 WRITE(age.c_str(), age.length()); // GCOVR_EXCL_LINE
274 } else {
275 uint64_t expires = cookie.second->getExpires();
276 if (expires > 0) {
277 WRITE("; Expires=", 10); // GCOVR_EXCL_LINE
278 auto date = DateTime(expires * 1000000, 0);
279 auto date_string = sese::text::DateTimeFormatter::format(date, TIME_GREENWICH_MEAN_PATTERN);
280 WRITE(date_string.c_str(), date_string.size()); // GCOVR_EXCL_LINE
281 }
282 }
283
284 bool secure = cookie.second->isSecure();
285 if (secure) {
286 WRITE("; Secure", 8); // GCOVR_EXCL_LINE
287 }
288
289 bool http_only = cookie.second->isHttpOnly();
290 if (http_only) {
291 WRITE("; HttpOnly", 10); // GCOVR_EXCL_LINE
292 }
293
294 WRITE("\r\n", 2); // GCOVR_EXCL_LINE
295 }
296 return true;
297}

329Cookie::Ptr HttpUtil::parseFromSetCookie(const std::string &text) noexcept {
330 auto result = StringBuilder::split(text, "; ");
331 Cookie::Ptr cookie = nullptr;
332 bool is_first = true;
333 for (decltype(auto) one: result) {
334 auto index = one.find('=');
335 if (is_first) {
336 if (index == std::string::npos) return nullptr;
337 cookie = std::make_shared<Cookie>(one.substr(0, index), one.substr(index + 1));
338 is_first = false;
339 } else {
340 if (index != std::string::npos) {
341 // 键值对
342 if (COMPARE(one.substr(0, index), "max-age")) {
343 cookie->setMaxAge(std::stoll(one.substr(index + 1)));
344 } else if (COMPARE(one.substr(0, index), "expires")) {
345 cookie->setExpires(text::DateTimeFormatter::parseFromGreenwich(one.substr(index + 1)));
346 } else if (COMPARE(one.substr(0, index), "path")) {
347 cookie->setPath(one.substr(index + 1));
348 } else if (COMPARE(one.substr(0, index), "domain")) {
349 cookie->setDomain(one.substr(index + 1));
350 }
351 } else {
352 // 属性
353 if (COMPARE(one, "secure")) {
354 cookie->setSecure(true);
355 } else if (COMPARE(one, "httponly")) {
356 cookie->setHttpOnly(true);
357 }
358 }
359 }
360 }
361
362 return cookie;
363}

367CookieMap::Ptr HttpUtil::parseFromCookie(const std::string &text) noexcept {
368 auto cookies = std::make_shared<CookieMap>();
369 auto result = StringBuilder::split(text, "; ");
370 for (decltype(auto) one: result) {
371 auto pair = StringBuilder::split(one, "=");
372 if (pair.size() != 2) continue;
373 cookies->add(std::make_shared<Cookie>(pair[0], pair[1]));
374 }
375 return cookies;
